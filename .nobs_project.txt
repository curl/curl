name curl

tools {
  obj2rsp_windows {
    tool_type = "command"
    WINDOWS
    !OBJ2RSP
    !OBJ
    !RSP_DIR
    !RSP_FILE

    cmdline = [
      "if not exist ${RSP_DIR} mkdir ${RSP_DIR}",
      "type nul > ${RSP_FILE}",
      "echo #{OBJ}>>${RSP_FILE}"
    ]
  }

  obj2rsp_linux {
    tool_type = "command"
    LINUX
    !OBJ2RSP
    !OBJ
    !RSP_DIR
    !RSP_FILE

    cmdline = [
      "mkdir -p ${RSP_DIR}",
      ": > ${RSP_FILE}",
      "printf '%s\n' #{OBJ} >> ${RSP_FILE}"
    ]
  }
}
