name curl

tools
{
  obj2rsp
  {
    tool_type = "command"
    !OBJ2RSP
    !OBJ
    !RSP_DIR
    !RSP_FILE

    cmdline = []
    DQUOTE = R"(")"
    PRINTF_FORMAT = R"('%s\n')"

    if os_family = "windows"
    {
      cmdline += [
        "if not exist ${DQUOTE}${RSP_DIR}${DQUOTE} mkdir ${DQUOTE}${RSP_DIR}${DQUOTE}",
        "type nul > ${DQUOTE}${RSP_FILE}${DQUOTE}",
        "echo #{OBJ}>>${DQUOTE}${RSP_FILE}${DQUOTE}"
      ]
    }

    if os_family = "unix"
    {
      cmdline += [
        "mkdir -p ${DQUOTE}${RSP_DIR}${DQUOTE}",
        ": > ${DQUOTE}${RSP_FILE}${DQUOTE}",
        "printf ${PRINTF_FORMAT} ${DQUOTE}#{OBJ}${DQUOTE} >> ${DQUOTE}${RSP_FILE}${DQUOTE}"
      ]
    }
  }
}
